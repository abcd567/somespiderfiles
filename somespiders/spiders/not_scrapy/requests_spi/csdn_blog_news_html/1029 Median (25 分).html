#include <cstdio>
#include <vector>
using namespace std;
int main()
{
    int cnt = 0, mid, N, M, iA = 0;
    scanf("%d", &N);
    vector<int> A;
    for(int i = 0, temp; i < N; ++i)
    {
        scanf("%d", &temp);
        A.push_back(temp);
    }
    scanf("%d", &M);
    mid = ( N + M - 1 ) / 2;
    for( int i = 0, temp; i < M; ++i )
    {
        scanf("%d", &temp);
        for( iA; iA < A.size() && A[iA] < temp; ++iA, ++cnt )
            if( cnt == mid )
            {
                printf("%d", A[iA]);
                return 0;
            }
        if( cnt++ == mid )
        {
            printf("%d", temp);
            return 0;
        }
    }
    printf("%d", A[iA + mid - cnt]);
}
